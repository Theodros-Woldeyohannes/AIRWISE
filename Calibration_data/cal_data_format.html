for (x in seq_along(my_data)) {
  df <- my_data[[x]]
  df$timestamp_clean <- sub("z$", "+0000", df$time)
  df$timestamp_datetime <- as.POSIXct(df$timestamp_clean, format = "%Y/%m/%dT%H:%M:%S%z", tz = "UTC")
  my_data[[x]] <- df
